// 1. You must create an Edge Function in your Supabase dashboard and paste this code in it. Let's assume it's called "send-push".
// 
// 2. Set the Secrets in your Supabase Edge Function settings:
//    - VAPID_PRIVATE_KEY
//    - VAPID_PUBLIC_KEY
//    - VAPID_SUBJECT (e.g. mailto:your-email@example.com)
// 
// 3. IMPORTANT: YOU MUST CREATE A DATABASE WEBHOOK!
//    - Go to Supabase Dashboard -> Database -> Webhooks
//    - Click "Create Webhook"
//    - Name: "Send Push on Message"
//    - Table: "messages"
//    - Events: Check "Insert"
//    - Type: "Supabase Edge Function" 
//    - Method: POST
//    - Edge Function: Select "send-push" (or whatever you named the function)
//    - Save the Webhook.
// 
// Without the webhook, this file does absolutely nothing because it will never be called when someone sends a message!

import { serve } from 'https://deno.land/std@0.168.0/http/server.ts'
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'
import webpush from 'https://esm.sh/web-push@3.6.7'

console.log("Hello from send-push edge function!")

const VAPID_PUBLIC_KEY = Deno.env.get('VAPID_PUBLIC_KEY')?.trim();
const VAPID_PRIVATE_KEY = Deno.env.get('VAPID_PRIVATE_KEY')?.trim();
let VAPID_SUBJECT = Deno.env.get('VAPID_SUBJECT')?.trim() || 'mailto:brare.black@gmail.com';
if (!VAPID_SUBJECT.startsWith('mailto:')) {
  VAPID_SUBJECT = 'mailto:' + VAPID_SUBJECT;
}

if (VAPID_PUBLIC_KEY && VAPID_PRIVATE_KEY) {
  console.log(`Initializing VAPID with public key starting with: ${VAPID_PUBLIC_KEY.substring(0, 10)}...`);
  webpush.setVapidDetails(
    VAPID_SUBJECT,
    VAPID_PUBLIC_KEY,
    VAPID_PRIVATE_KEY
  );
} else {
  console.error("CRITICAL: VAPID keys are missing from Edge Function Secrets!");
}

serve(async (req) => {
  console.log("----- INCOMING REQUEST TRIGGERED -----");
  
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: { 'Access-Control-Allow-Origin': '*' } });
  }

  try {
    const rawBody = await req.text();
    console.log("Raw payload:", rawBody);
    
    let record;
    try {
      record = JSON.parse(rawBody);
    } catch (e) {
      console.error("Failed to parse JSON payload");
      return new Response('Invalid JSON', { status: 400 });
    }
    
    // Webhook shape usually is: { type: 'INSERT', record: { ... }, table: 'messages' }
    const message = record.record || record;
    console.log("Parsed message:", message);
    
    if (!message || !message.receiver_id) {
      console.warn("Message missing receiver_id.");
      return new Response('No valid message structure found', { status: 400 });
    }

    // Initialize Supabase Client
    const supabaseUrl = Deno.env.get('SUPABASE_URL');
    const supabaseServiceKey = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY');
    
    if (!supabaseUrl || !supabaseServiceKey) {
      console.error("Missing SUPABASE_URL or SUPABASE_SERVICE_ROLE_KEY environment variables!");
    }
    
    const supabase = createClient(supabaseUrl!, supabaseServiceKey!);

    console.log(`Fetching subscriptions for user: ${message.receiver_id}`);
    const { data: subscriptions, error: subError } = await supabase
      .from('push_subscriptions')
      .select('*')
      .eq('user_id', message.receiver_id);

    if (subError) {
      console.error("Error fetching subscriptions:", subError);
    }

    if (!subscriptions || subscriptions.length === 0) {
      console.log(`No active subscriptions found for user: ${message.receiver_id}`);
      return new Response('No subscriptions found', { status: 200 });
    }
    
    console.log(`Found ${subscriptions.length} subscriptions. Proceeding to push...`);

    // Deduplicate subscriptions by endpoint
    const uniqueSubs = new Map();
    for (const sub of subscriptions) {
      uniqueSubs.set(sub.endpoint, sub);
    }
    const dedupedSubscriptions = Array.from(uniqueSubs.values());
    console.log(`After deduping, sending to ${dedupedSubscriptions.length} endpoints.`);

    // Fetch the sender profile for the name
    const { data: senderProf } = await supabase
      .from('profiles')
      .select('full_name, username')
      .eq('id', message.sender_id)
      .single();

    const senderName = senderProf?.full_name || senderProf?.username || 'Someone';

    const payload = JSON.stringify({
      title: senderName,
      body: message.content || "Sent you a media message",
      data: {
        url: "/" // needed for service worker
      }
    });

    const pushPromises = dedupedSubscriptions.map((sub: any) => {
      const pushSubscription = {
        endpoint: sub.endpoint,
        keys: {
          auth: sub.auth_key,
          p256dh: sub.p256dh_key
        }
      };
      
      const options = { 
        urgency: 'high',
        TTL: 86400,
        topic: 'socium-message'
      };
      
      return webpush.sendNotification(pushSubscription, payload, options)
        .then((response: any) => {
           console.log(`Push sent successfully to ${sub.endpoint.substring(0, 30)}... | Status: ${response.statusCode} | Headers: ${JSON.stringify(response.headers)} | Body: ${response.body}`);
        })
        .catch(async (err: any) => {
           if (err.statusCode === 404 || err.statusCode === 410) {
              console.log(`Subscription expired/invalid for endpoint: ${sub.endpoint}, removing from DB.`);
              await supabase.from('push_subscriptions').delete().eq('endpoint', sub.endpoint);
           } else {
              console.error(`Push delivery error for ${sub.endpoint}: statusCode=${err.statusCode}, headers=${JSON.stringify(err.headers)}, body=${err.body}`);
              console.error(err);
           }
        });
    });

    await Promise.all(pushPromises);
    console.log("All push promises settled.");

    return new Response(JSON.stringify({ success: true }), { headers: { 'Content-Type': 'application/json' } });
  } catch (error: any) {
    console.error("Fatal error in edge function:", error);
    return new Response(JSON.stringify({ error: error.message }), { status: 500 });
  }
})
