// 1. You must create an Edge Function in your Supabase dashboard and paste this code in it. Let's assume it's called "send-push".
// 
// 2. Set the Secrets in your Supabase Edge Function settings:
//    - FIREBASE_SERVICE_ACCOUNT (This should be the raw JSON string of your Firebase Service Account JSON file)
//    - FIREBASE_DATABASE_URL (This is your Firebase Realtime Database URL)
//      You can get this from Firebase Console -> Project Settings -> Service Accounts -> "Generate new private key"
// 
// 3. IMPORTANT: Delete any existing "Send Push on Message" Database Webhooks!
//    - Go to Supabase Dashboard -> Database -> Webhooks
//    - Find the old 'Send Push on Message' webhook.
//    - DELETE it! The app now invokes the edge function manually from the frontend
//      client to ensure presence state and badge unread counts sync correctly.
// 
// Without updating the edge function code and deleting the old webhook, the notifications might trigger twice or fail.

import { serve } from 'https://deno.land/std@0.168.0/http/server.ts'
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'
import admin from 'npm:firebase-admin@11'

// Initialize Firebase Admin if not already initialized
const serviceAccountRaw = Deno.env.get('FIREBASE_SERVICE_ACCOUNT');
const databaseURL = Deno.env.get('FIREBASE_DATABASE_URL');

if (serviceAccountRaw && databaseURL) {
  try {
    const serviceAccount = JSON.parse(serviceAccountRaw);
    if (!admin.apps.length) {
      admin.initializeApp({
        credential: admin.credential.cert(serviceAccount),
        databaseURL: databaseURL
      });
      console.log("Firebase Admin initialized successfully.");
    }
  } catch (error) {
    console.error("CRITICAL: Failed to parse FIREBASE_SERVICE_ACCOUNT JSON or initialize admin.", error);
  }
} else {
  console.error("CRITICAL: FIREBASE_SERVICE_ACCOUNT or FIREBASE_DATABASE_URL secret is missing from Edge Function Secrets!");
}

export const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
};

serve(async (req) => {
  console.log("----- INCOMING REQUEST TRIGGERED -----");
  
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders });
  }

  if (!admin.apps.length) {
    console.error("Firebase Admin is not configured. Aborting push delivery.");
    return new Response('Firebase Admin not initialized', { status: 500, headers: corsHeaders });
  }

  try {
    const rawBody = await req.text();
    console.log("Raw payload:", rawBody);
    
    let record;
    try {
      record = JSON.parse(rawBody);
    } catch (e) {
      console.error("Failed to parse JSON payload");
      return new Response('Invalid JSON', { status: 400, headers: corsHeaders });
    }
    
    // Webhook shape usually is: { type: 'INSERT', record: { ... }, table: 'messages' }
    if (record.type === 'INSERT' && record.table === 'messages') {
      console.log("Webhook trigger detected. Ignoring because client handles push and RTDB sync.");
      return new Response('Webhook ignored to prevent duplicate push', { status: 200, headers: corsHeaders });
    }
    
    const message = record.record || record;
    console.log("Parsed message:", message);
    
    if (!message || !message.receiver_id) {
      console.warn("Message missing receiver_id.");
      return new Response('No valid message structure found', { status: 400, headers: corsHeaders });
    }

    // Initialize Supabase Client
    const supabaseUrl = Deno.env.get('SUPABASE_URL');
    const supabaseServiceKey = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY');
    
    const supabase = createClient(supabaseUrl!, supabaseServiceKey!);

    try {
      const presenceSnapshot = await admin.database().ref(`global_presence/${message.receiver_id}`).once('value');
      if (presenceSnapshot.val() === true) {
        console.log(`Recipient ${message.receiver_id} is online. Skipping push.`);
        return new Response('User is online, push skipped', { status: 200, headers: corsHeaders });
      }
    } catch (e) {
      console.error("Failed to check presence in edge function. Proceeding...", e);
    }

    console.log(`Fetching FCM tokens for user: ${message.receiver_id}`);
    const { data: subscriptions, error: subError } = await supabase
      .from('push_subscriptions')
      .select('*')
      .eq('user_id', message.receiver_id);

    if (subError) {
      console.error("Error fetching subscriptions:", subError);
    }

    if (!subscriptions || subscriptions.length === 0) {
      console.log(`No active subscriptions found for user: ${message.receiver_id}`);
      return new Response('No subscriptions found', { status: 200, headers: corsHeaders });
    }
    
    // Deduplicate endpoints (tokens)
    const uniqueSubs = new Map();
    for (const sub of subscriptions) {
      uniqueSubs.set(sub.endpoint, sub);
    }
    const dedupedSubscriptions = Array.from(uniqueSubs.values());
    console.log(`Sending to ${dedupedSubscriptions.length} FCM endpoints.`);

    // Fetch the sender profile for the name
    const { data: senderProf } = await supabase
      .from('profiles')
      .select('full_name, username')
      .eq('id', message.sender_id)
      .single();

    const senderName = senderProf?.full_name || senderProf?.username || 'Someone';

    let bodyText = message.content;
    if (message.media_type === 'shared_post') {
      bodyText = 'Shared a post with you';
    } else if (!bodyText) {
      if (message.media_type === 'image') {
        bodyText = 'Sent you an image';
      } else if (message.media_type === 'audio') {
        bodyText = 'Sent you a voice message';
      } else if (message.media_type === 'location') {
        bodyText = 'Sent you a location';
      } else {
        bodyText = 'Sent you a media message';
      }
    }

    const pushPromises = dedupedSubscriptions.map(async (sub: any) => {
      const fcmToken = sub.endpoint;
      
      // Calculate badge count from Firebase Realtime Database
      let badgeCount = 0;
      try {
        const snapshot = await admin.database().ref(`unseen_chat_count/${message.receiver_id}`).once('value');
        badgeCount = snapshot.val() || 0;
      } catch (dbError) {
        console.error("Failed to read unseen_chat_count from RTDB, defaulting to 0", dbError);
      }

      const payload = {
        token: fcmToken,
        notification: {
          title: senderName,
          body: bodyText
        },
        apns: {
          payload: {
            aps: {
              badge: badgeCount,
              sound: 'default'
            }
          }
        },
        data: {
          title: senderName,
          body: bodyText,
          url: `/?chatId=${message.sender_id}`,
          senderId: message.sender_id || "",
          badge: String(badgeCount)
        }
      };
      
      try {
        const response = await admin.messaging().send(payload);
        console.log(`Firebase push sent successfully to token ${fcmToken.substring(0, 20)}... | Message ID: ${response}`);
      } catch (err: any) {
        console.error(`Firebase push error for token ${fcmToken.substring(0, 20)}...:`, err);
        
        // Cleanup invalid FCM tokens
        if (
          err.code === 'messaging/invalid-registration-token' ||
          err.code === 'messaging/registration-token-not-registered' ||
          err.code === 'messaging/invalid-argument'
        ) {
          console.log(`Token is invalid or unregistered. Removing from DB...`);
          await supabase.from('push_subscriptions').delete().eq('endpoint', fcmToken);
        }
      }
    });

    await Promise.all(pushPromises);
    console.log("All push promises settled.");

    return new Response(JSON.stringify({ success: true }), { headers: { ...corsHeaders, 'Content-Type': 'application/json' } });
  } catch (error: any) {
    console.error("Fatal error in edge function:", error);
    return new Response(JSON.stringify({ error: error.message }), { status: 500, headers: corsHeaders });
  }
})
